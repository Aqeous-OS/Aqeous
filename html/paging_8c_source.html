216 void page_fault(registers_t regs)
217 {
218  asm volatile("cli");
219  // A page fault has occurred.
220  // The faulting address is stored in the CR2 register.
221  uint32_t faulting_address;
222  asm volatile("mov %%cr2, %0" : "=r" (faulting_address));
223 
224  // The error code gives us details of what happened.
225  int present = !(regs.err_code & 0x1); // Page not present
226  int rw = regs.err_code & 0x2; // Write operation?
227  int us = regs.err_code & 0x4; // Processor was in user-mode?
228  int reserved = regs.err_code & 0x8; // Overwritten CPU-reserved bits of page entry?
229  int id = regs.err_code & 0x10; // Caused by an instruction fetch?
230 
231  // Output an error message.
232  console_writestring("\nPage fault! ( ");
233  if (present)
234  {
235  console_writestring("present, Allocating page for it ");
236  MapPage((void*)faulting_address,(void*)faulting_address);
237  }
238  if (rw)
239  {
240  console_writestring("read-only ");
241  }
242  if (us) {console_writestring("user-mode ");}
243  if (reserved) {console_writestring("reserved ");}
244 
245  if (id) {console_writestring("id "); console_write_dec(id);}
246  console_writestring(") at 0x");
247  console_write_dec(faulting_address);
248  console_writestring(" - EIP: ");
249  console_write_dec(regs.eip);
250  console_writestring("\n");
251  asm volatile(" \
252  cli; \
253  mov %0, %%ecx; \
254  mov %1, %%esp; \
255  mov %2, %%ebp; \
256  sti; \
257  jmp *%%ecx "
258  : : "r"(regs.eip), "r"(regs.esp), "r"(regs.ebp));
259  asm volatile("sti");
260  asm volatile("iret");
261  // PANIC("Page fault");
262 }
